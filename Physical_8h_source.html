67  static double radiusEarth(
68  const double latitude = std::numeric_limits<double>::max()) {
69  if (latitude == std::numeric_limits<double>::max()) return 6378206.40;
70  const double l = deg2rad() * latitude;
71  return std::sqrt(
72  (std::pow(equatorialRadius(), 4.0) * std::cos(l) * std::cos(l) +
73  std::pow(polarRadius(), 4.0) * std::sin(l) * std::sin(l)) /
74  (std::pow(equatorialRadius(), 2.0) * std::cos(l) * std::cos(l) +
75  std::pow(polarRadius(), 2.0) * std::sin(l) * std::sin(l)));
76  }

87  static double geodesic_distance(const double x1, const double y1,
88  const double x2, const double y2) {
89  const double lat1 = deg2rad() * y1;
90  const double lon1 = deg2rad() * x1;
91  const double lat2 = deg2rad() * y2;
92  const double lon2 = deg2rad() * x2;
93  return 2.0 * radiusEarth(y1, y2) *
94  std::asin(
95  std::sqrt(std::pow(std::sin((lat2 - lat1) / 2.0), 2.0) +
96  std::cos(lat1) * std::cos(lat2) *
97  std::pow(std::sin((lon2 - lon1) / 2.0), 2.0)));
98  }

115  static double azimuthEarth(const double x1, const double y1, const double x2,
116  const double y2) {
117  const double dx = deg2rad() * radiusEarth(y1, y2) * (x1 - x2) *
118  std::cos(deg2rad() * (y2));
119  const double dy = deg2rad() * radiusEarth(y1, y2) * (y1 - y2);
120  const double azi = 360.0 + rad2deg() * std::atan2(dy, dx);
121  return azi > 360.0 ? azi - 360.0 : azi;
122  }
