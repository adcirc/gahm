74 static double radiusEarth(
75  const double latitude = std::numeric_limits<double>::max()) {
76  if (latitude == std::numeric_limits<double>::max()) return 6378206.40;
77  const double l = deg2rad() * latitude;
78  return std::sqrt(
79  (std::pow(equatorialRadius(), 4.0) * std::cos(l) * std::cos(l) +
80  std::pow(polarRadius(), 4.0) * std::sin(l) * std::sin(l)) /
81  (std::pow(equatorialRadius(), 2.0) * std::cos(l) * std::cos(l) +
82  std::pow(polarRadius(), 2.0) * std::sin(l) * std::sin(l)));
83 }

94 static double geodesic_distance(const double x1, const double y1,
95  const double x2, const double y2) {
96  const double lat1 = deg2rad() * y1;
97  const double lon1 = deg2rad() * x1;
98  const double lat2 = deg2rad() * y2;
99  const double lon2 = deg2rad() * x2;
100  return 2.0 * radiusEarth(y1, y2) *
101  std::asin(std::sqrt(std::pow(std::sin((lat2 - lat1) / 2.0), 2.0) +
102  std::cos(lat1) * std::cos(lat2) *
103  std::pow(std::sin((lon2 - lon1) / 2.0), 2.0)));
104 }

123 static double azimuthEarth(const double x1, const double y1, const double x2,
124  const double y2) {
125  const double lam0 = x1 * Physical::deg2rad();
126  const double phi0 = y1 * Physical::deg2rad();
127  const double lam1 = x2 * Physical::deg2rad();
128  const double phi1 = y2 * Physical::deg2rad();
129  const double dlam = lam1 - lam0;
130  const double a = std::sin(dlam) * std::cos(phi1);
131  const double b = std::cos(phi0) * std::sin(phi1);
132  const double c = std::sin(phi0) * std::cos(phi1) * std::cos(dlam);
133  const double azi = std::atan2(a, b - c);
134  return azi * Physical::rad2deg();
135 }

150 static constexpr double frictionAngle(double r, double rmx) noexcept {
151  if (0.0 < r && r < rmx) {
152  return 10.0 * r / rmx;
153  } else if (rmx <= r && r < 1.2 * rmx) {
154  return 10.0 + 75.0 * (r / rmx - 1.0);

166 template <typename T>
167 static constexpr T fast_exp(const T x) noexcept {
168  if (std::is_same<T, float>()) {
169  constexpr auto v0 = double((1 << 20) / M_LN2);
170  constexpr auto v1 = double((1 << 20) * 1023 - 0);
171  union union_exp {
172  double d_;
173  int32_t i_[2];
174  explicit union_exp(int32_t v) : i_{0, v} {}
175  };
176  union_exp uu(v0 * x + v1);
177  return uu.d_;
178  } else if (std::is_same<T, double>()) {

185  };

187  return uu.d_;
